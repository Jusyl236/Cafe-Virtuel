analyse_logs.py
Process logs from v2_runs_20251220 to compute aggregated metrics and generate canonical figures.
Input: logs/grok_hardware_sim/v2_runs_20251220/*.csv
Outputs: metrics_table.csv, figures (entropy_evolution.png, variance_decay.png, final_state_histogram.png)

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from pathlib import Path
import seaborn as sns
from scipy.stats import entropy as scipy_entropy
from scipy.stats import pearsonr
import warnings
warnings.filterwarnings('ignore')

# ==================== CONFIGURATION ====================
LOG_DIR = Path("logs/grok_hardware_sim/v2_runs_20251220")
OUTPUT_DIR = Path("analysis_output")
OUTPUT_DIR.mkdir(exist_ok=True)

MODELS = ["mem4ristor", "kuramoto", "dist_averaging", "voter"]
N_RUNS = 50
N_AGENTS = 400
STEPS = 500
VARIANCE_RATIO_THRESHOLD = 0.20  # for tau80

# ==================== METRICS COMPUTATION ====================

def compute_entropy(opinions, bins=10):
    """Shannon entropy of discretized opinions in [-1, 1]."""
    hist, _ = np.histogram(opinions, bins=bins, range=(-1, 1))
    prob = hist / np.sum(hist)
    prob = prob[prob > 0]
    return -np.sum(prob * np.log2(prob))

def compute_tau80(variance_series, initial_variance):
    """First step where variance < 0.20 * initial_variance."""
    threshold = VARIANCE_RATIO_THRESHOLD * initial_variance
    below = np.where(variance_series <= threshold)[0]
    return below[0] if len(below) > 0 else len(variance_series) - 1

def process_single_run(df_run):
    """Compute metrics for one run."""
    model = df_run['model'].iloc[0]
    run_id = df_run['run_id'].iloc[0]
    
    # Extract time series data
    steps = df_run['step'].unique()
    variance_series = []
    entropy_series = []
    
    for step in steps:
        step_opinions = df_run[df_run['step'] == step]['opinion'].values
        variance_series.append(np.var(step_opinions))
        entropy_series.append(compute_entropy(step_opinions))
    
    variance_series = np.array(variance_series)
    entropy_series = np.array(entropy_series)
    
    # Key metrics
    initial_variance = variance_series[0]
    final_opinions = df_run[df_run['step'] == steps[-1]]['opinion'].values
    H_final = compute_entropy(final_opinions)
    tau80 = compute_tau80(variance_series, initial_variance)
    
    # Bias correlation (bias is constant per run, taken from first row)
    bias_value = df_run['bias_input'].iloc[0]
    bias_vector = np.full_like(final_opinions, bias_value)
    corr, _ = pearsonr(final_opinions, bias_vector)
    bias_corr_abs = abs(corr)
    
    # Diversit√© D√©lib√©rative
    DD = H_final * tau80 * (1 - bias_corr_abs)
    
    return {
        'model': model,
        'run_id': run_id,
        'H_final': H_final,
        'tau80': tau80,
        'bias_corr_abs': bias_corr_abs,
        'DD': DD,
        'final_variance': variance_series[-1],
        'final_mean_opinion': np.mean(final_opinions),
        'variance_series': variance_series,
        'entropy_series': entropy_series,
        'final_opinions': final_opinions
    }
